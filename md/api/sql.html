    目前存储使用 mysql8.* 作为后端存储，目前存在的数据库表为

    数据库表名字 描述
    language 多语言表
    books 书籍信息表
    categories 分类表
    bookcategories 书籍分类表
    chapters 章节表

    # language

    多语言表

    # 属性解析

    属性名 类型 描述
    id - 主键
    language str 语言缩写，前后端语言标识
    display str 前端语言显示
    description str 注释
    sort_number int 排序
    active int 是否激活

    # 索引

    无

    # 数据

    查看具体数据 
    ZH = "ZH"  # 中文(Chinese)
EN = "EN"  # 英语(English)
JA = "JA"  # 日语(Japanese)
KO = "KO"  # 韩语(Korean)
TH = "TH"  # 泰语(Thai)

    # books

    书籍表

    # 属性解析

    属性名 类型 描述
    id - 主键
    title str 小说名称
    title_ZH str 中文小说名称
    isbn str 小说的 IDBN 编号
    author str 小说作者
    up_id str 上传者的编号，64位
    language str 语言标识
    book64 str 该本小说的唯一标识，64位
    subbook64 str 该小说指定唯一标识，64位
    description str 小说简介
    description_ZH str 中文小说简介
    cover_url str 小说封面
    categories int 小说章节数量
    translate_categories int 翻译章节数量
    status str 小说状态
    translate_status str 翻译状态
    active int 小说激活状态
    country str 小说国家
    is_original int 是否是原版
    resources str 书籍来源
    created_at date 小说创建时间
    updated_at date 小说更新时间

    # status 属性解析

    查看具体数据 
    SER = "SER"  # 连载中
COM = "COM"  # 已完结
HIA = "HIA"  # 暂停更新
DIS = "DIS"  # 断更

    # translate_status 属性解析

    查看具体数据 
    SER = "SER"  # 翻译中
COM = "COM"  # 翻译完成
HIA = "HIA"  # 暂停翻译

    # contry 属性解析

    查看具体数据 
    CN = "中国"

    # active 属性解析

    查看具体数据 
    PUB = 1  # 公开
DRA = 2  # 草稿
DEL = 3  # 删除
OFF = 4  # 下架

    # categories 分类表

    分类表

    # 属性解析

    属性名 类型 描述
    id - 主键
    category str 一级分类名
    category_int int 一级分类对应int
    sub_category str 二级分类名
    sub_category_int int 二级分类对应int
    active int 分类状态，为 1 的时候分类才正式启用
    sort_number int 一级分类排列编码
    sub_sort_number int 二级分类排列编码
    language str 语言标识
    description str 分类描述

    # bookcategories 书籍分类表

    书籍分类表，是一个完全外键表。

    属性名 类型 描述
    id - 主键
    book_id 外键 对应books表中的id
    category_id 外键 对应categories表中的id

    # chapters 章节表

    章节表。

    属性名 类型 描述
    id - 主键
    subbook64 str 对应书籍表中的subbook64
    title str 章节名
    content blob 章节内容，Blob格式存储
    sort_number int 章节排序，从 0 开始
    chapter_title str 卷名
    chapter_number int 卷排序，从 0 开始
    chapter_sort_number int 卷章排序，从 0 开始
    display_level int 章节展示标识
    created_at date 章节创建时间
    updated_at date 章节更新时间

    # display_level 属性解析

    查看具体数据 
    FRE = 1  # 免费试读
LOGIN = 2  # 注册章节
VIP = 3  # VIP章节
SUPER = 4  # 购买本书
